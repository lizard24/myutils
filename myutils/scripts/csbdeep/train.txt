
NOTES:

-  "--file_npz" has to be set
-  training results are saved in a subfolder (=="--model_name") in "my_model"


OPTIONS:

********* DATA

"--file_npz"			default=None		type=str	help="Npz file with training data."
"--val_split"			default=0.1		type=float	help="Percentage of validation split."

"--folder_data"			default=None		type=str	help="Alternative to 'file_npz'. 'folder_data' is directory to subfolders that contain training data ('input', target', 'target_val', 'input_val')."
"--axes"			default='SYXC'		type=str	help="Axes of training data. Needs to be defined if 'file_npz' is None."

********* TRAINING

"--model_folder"		default='models'	type=str	help="Folder the model is saved in."

"--model_name"			default='my_model'	type=str	help="Name of the folder the model is saved in."

"--train_loss"			default="mae"		type=str	help="Loss function. Select between 'mae', 'mse', 'ssim', 'ms_ssim', 'probabilistic'"

"--ms_ssim_no_weights"		default=3		type=int	help="Number of weights if train_loss is 'ms_ssim'."

"--ms_ssim_filter_size"		default=11		type=int	help="Filter size of Gaussian window if train_loss is 'ms_ssim'."

"--train_learning_rate"		default=5e-3		type=float	help="Learning rate"

"--train_batch_size"		default=4		type=int	help="Batch size"

"--train_steps_per_epoch"	default=50		type=int	help="Steps per epoch"

"--train_epochs"		default=50		type=int	help="Epoch number"

"--callback_period"		default=None		type=int	help='If/how often weights are saved in folder "callbacks".'


********* CONTINUE PREVIOUS TRAINING

"--continue_training"		action='store_true'			help="Continue training with weights from last callback if existent. Default: False"

"--model_name_pretrained"	default=None		type=str	help="Folder to a pretrained model."
"--train_epochs_pretrained"	default=0		type=int	help='Last epoch of model_name_pretrained.'


********* OTHER

"--display"			default=False		type=boolean	help="Display info yes/no"
